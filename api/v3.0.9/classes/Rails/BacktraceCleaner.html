def initialize
  super
  add_filter   { |line| line.sub("#{Rails.root}/", '') }
  add_filter   { |line| line.sub(RENDER_TEMPLATE_PATTERN, '') }
  add_filter   { |line| line.sub('./', '/') } # for tests

  add_gem_filters
  add_silencer { |line| line !~ APP_DIRS_PATTERN }
end
