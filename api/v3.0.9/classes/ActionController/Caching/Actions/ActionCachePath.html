def initialize(controller, options = {}, infer_extension = true)
  if infer_extension
    @extension = controller.params[:format]
    options.reverse_merge!(:format => @extension) if options.is_a?(Hash)
  end

  path = controller.url_for(options).split(%{://}).last
  @path = normalize!(path)
end
