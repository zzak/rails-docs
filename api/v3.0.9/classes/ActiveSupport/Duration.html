ActiveSupport::Duration
In: activesupport/lib/active_support/duration.rb:
Parent: ActiveSupport::BasicObject

Provides accurate date and time measurements using Date#advance and Time#advance, respectively. It mainly supports the methods on Numeric. Example:

1.month.ago       # equivalent to Time.now.advance(:months => -1)

def +(other)
  if Duration === other
    Duration.new(value + other.value, @parts + other.parts)
  else
    Duration.new(value + other, @parts + [[:seconds, other]])
  end
end

def ==(other)
  if Duration === other
    other.value == value
  else
    other == value
  end
end
