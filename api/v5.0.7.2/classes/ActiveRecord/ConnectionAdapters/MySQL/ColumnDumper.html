def column_spec_for_primary_key(column)
  if column.bigint?
    spec = { id: :bigint.inspect }
    spec[:default] = schema_default(column) || 'nil' unless column.auto_increment?
  else
    spec = super
  end
  spec[:unsigned] = 'true' if column.unsigned?
  spec
end
